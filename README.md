## Comandos Aprendidos

pwd muestra donde estoy ubicado 
ls lista archivos
ls -la muestra archivos ocultos y permisos


## Estructura

/home 	Usuarios
/etc 	Configuracion
/var 	Logs y datos variables
/usr 	Programas
/bin 	Comandos básicos
/opt	Software adicional
/tmp	Archivos temporales


## Comandos escenciales
cd ~ 	Me ubica en inicio
cd ..	Me ubica en la carpeta madre
cd / 	Me ubica en la carpeta hijo
mkdir	Crea directorios
touch	Crea archivos
nano	Edita archivos
ls -l	Muestra permisos
tree	Ver carpetas y subcarpetas


chmod 644 "Nombre_archivo"		Cambia permisos
sudo chown usuario "Nombre_archivo"	Cambia propietario

ps aux 	Procesos
top 	Ver procesos
kill 	Matar proceso

sudo apt update
sudo apt upgrade


## Variables de entorno
export MI_VAR= "Nombre-variable"	Crea variable
echo MI_VAR				Muestra nombre variable
